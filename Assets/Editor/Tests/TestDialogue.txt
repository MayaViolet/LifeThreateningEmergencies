define annie = Character('Annie', 'annie_portrait')
define varus = Character("Varus")

label start:
	"Ambient text ..."

	annie "Something about rice cakes."
	varus "Sounds good."
	
menu:
	"Thin rice cakes.":
		jump thin
	
	"Thick rice cakes.":
		jump thick
		
label thin:
	annie "You're right, thin rice cakes are the best."
	$ thin = True
	jump target

label thick:
	varus "Excellent choice."
	$ thin = False
	jump target
	
target:
	annie "Here I am."
